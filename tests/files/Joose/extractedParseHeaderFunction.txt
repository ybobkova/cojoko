function (headers) {
        // copy von jquery
        var m, rheaders = /^(.*?):[ \t]*([^\r\n]*)\r?$/gm; // IE leaves an \r character at EOL
        this.setHeader({});
  
        while (m = rheaders.exec(headers)) {
          this.setHeaderField(m[1], m[2]);
        }

        return this;
      }
